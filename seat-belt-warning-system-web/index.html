// Smart Seat Belt Warning System - AVR CTC State Machine
#include <avr/io.h>
#include <avr/interrupt.h>

volatile int Count = 0;           // CTC timer tick accumulator
volatile int seat_occupied = 0;   // 1 when weight sensor active
volatile int seat_belt = 0;       // 1 when latch clip engaged
volatile int req_delay = 0;       // 1 after countdown expires

int main() {
    sei();                        // Global interrupt enable
    DDRD |= (1 << PD7);           // Set PD7 (Pin 7) as Output (LED/Buzzer)
    init_interrupts();            // Config INT0 and INT1 logical changes

    while(1) {
        if (seat_occupied == 1) {
            if (seat_belt == 0) {
                timer0_init();    // Initialize delay countdown (Timer0)
                if (req_delay == 1) {
                    PORTD |= (1 << PD7); // Trigger warning LED HIGH
                    pwm_on();            // Trigger warning buzzer (PWM)
                }
            } else {
                PORTD &= ~(1 << PD7); // Safety state: LED OFF
                timer_off();      // Stop Timer0 counting
                pwm_off();        // Silent Buzzer
            }
        } else {
            PORTD &= ~(1 << PD7); // Empty state: LED OFF
            timer_off();          // Stop Timer0 counting
            pwm_off();            // Silent Buzzer
        }
    }
}

Why is a Serial Monitor needed?

Microcontrollers are "headless" units without native screens. To transmit variables and debug state shifts, they stream binary data to a host PC using **UART Serial Transmit pins**.

The console on the left displays the exact string stream outputted by the C++ `Serial.println()` compiler commands when logical branches run in the firmware state-machine loop.
